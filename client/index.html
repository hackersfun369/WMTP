Protocol Overview

// Connect to WMTP server
const transport = new WebTransport("https://api.wmtp.online");
await transport.ready;

// Initialize session
send({ cmd: "INIT" });
// Response: { status: "OK", session_token: "WMTP-xxx" }

// Authenticate
send({ cmd: "AUTH", data: { email: "user@example.com" } });
// Response: { status: "OK", authenticated: true }

// Send mail (coming soon)
send({ cmd: "SEND", data: { to: "friend@example.com", body: "Hello!" } });
